# 1. Kolla fjärranslutningen (remote):
git remote -v

# 2. Kolla vilken gren du arbetar på:
git branch

# 3. Kolla om det finns ändringar att commit:a:
git status

# 4. Stage alla ändrade filer:
git add .

# 5. Skapa en commit:
git commit -m "Ditt meddelande här"

# 6. Skicka ändringarna till fjärrservern:
git push origin main
